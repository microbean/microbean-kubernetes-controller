054/**
055 * A pump of sorts that continuously "pulls" logical events out of
056 * Kubernetes and {@linkplain EventCache#add(Object, Event.Type,
057 * HasMetadata) adds them} to an {@link EventCache} so as to logically
058 * "reflect" the contents of Kubernetes into the cache.
059 *
060 * <h2>Thread Safety</h2>
061 *
062 * <p>Instances of this class are safe for concurrent use by multiple
063 * {@link Thread}s.</p>
064 *
065 * @param <T> a type of Kubernetes resource
066 *
067 * @author <a href="https://about.me/lairdnelson"
068 * target="_parent">Laird Nelson</a>
069 *
070 * @see EventCache
071 */
072@ThreadSafe
073public class Reflector<T extends HasMetadata> implements Closeable {

178  /**
179   * Creates a new {@link Reflector}.
180   *
181   * @param <X> a type that is both an appropriate kind of {@link
182   * Listable} and {@link VersionWatchable}, such as the kind of
183   * operation returned by {@link
184   * DefaultKubernetesClient#configMaps()} and the like
185   *
186   * @param operation a {@link Listable} and a {@link
187   * VersionWatchable} that can report information from a Kubernetes
188   * cluster; must not be {@code null}
189   *
190   * @param eventCache an {@link EventCache} <strong>that will be
191   * synchronized on</strong> and into which {@link Event}s will be
192   * logically "reflected"; must not be {@code null}
193   *
194   * @param synchronizationExecutorService a {@link
195   * ScheduledExecutorService} to be used to tell the supplied {@link
196   * EventCache} to {@linkplain EventCache#synchronize() synchronize}
197   * on a schedule; may be {@code null} in which case no
198   * synchronization will occur
199   *
200   * @param synchronizationInterval a {@link Duration} representing
201   * the time in between one {@linkplain EventCache#synchronize()
202   * synchronization operation} and another; may be {@code null} in
203   * which case no synchronization will occur
204   *
205   * @see #start()
206   */
207  @SuppressWarnings("rawtypes") // kubernetes-client's implementations of KubernetesResourceList use raw types
208  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Reflector(final X operation,
209                                                                                                                              final EventCache<T> eventCache,
210                                                                                                                              final ScheduledExecutorService synchronizationExecutorService,
211                                                                                                                              final Duration synchronizationInterval) {
212    super();
213    Objects.requireNonNull(operation);
214    this.eventCache = Objects.requireNonNull(eventCache);
215    // TODO: research: maybe: operation.withField("metadata.resourceVersion", "0")?    
216    this.operation = operation.withResourceVersion("0");
217    this.synchronizationInterval = synchronizationInterval;
218    if (synchronizationExecutorService == null) {
219      if (synchronizationInterval == null) {
220        this.synchronizationExecutorService = null;
221        this.shutdownSynchronizationExecutorServiceOnClose = false;
222      } else {
223        this.synchronizationExecutorService = Executors.newScheduledThreadPool(1);
224        this.shutdownSynchronizationExecutorServiceOnClose = true;
225      }
226    } else {
227      this.synchronizationExecutorService = synchronizationExecutorService;
228      this.shutdownSynchronizationExecutorServiceOnClose = false;
229    }
230  }

238  /**
239   * Notionally closes this {@link Reflector} by terminating any
240   * {@link Thread}s that it has started and invoking the {@link
241   * #onClose()} method while holding this {@link Reflector}'s
242   * monitor.
243   *
244   * @exception IOException if an error occurs
245   *
246   * @see #onClose()
247   */
248  @Override
249  public synchronized final void close() throws IOException {
250    try {
251      final ScheduledFuture<?> synchronizationTask = this.synchronizationTask;
252      if (synchronizationTask != null) {
253        synchronizationTask.cancel(false);
254      }
255      this.closeSynchronizationExecutorService();
256      if (this.watch != null) {
257        this.watch.close();
258      }
259    } finally {
260      this.onClose();
261    }
262  }
263
264  private synchronized final void closeSynchronizationExecutorService() {
265    if (this.synchronizationExecutorService != null && this.shutdownSynchronizationExecutorServiceOnClose) {
266      this.synchronizationExecutorService.shutdown();
267      try {
268        if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
269          this.synchronizationExecutorService.shutdownNow();
270          if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
271            this.synchronizationExecutorService.shutdownNow();
272          }
273        }
274      } catch (final InterruptedException interruptedException) {
275        this.synchronizationExecutorService.shutdownNow();
276        Thread.currentThread().interrupt();
277      }
278    }
279  }
280
281  private synchronized final void setUpSynchronization() {
282    if (this.synchronizationExecutorService != null) {
283      
284      final Duration synchronizationDuration = this.getSynchronizationInterval();
285      final long seconds;
286      if (synchronizationDuration == null) {
287        seconds = 0L;
288      } else {
289        seconds = synchronizationDuration.get(ChronoUnit.SECONDS);
290      }
291        
292      if (seconds > 0L) {
293        final ScheduledFuture<?> job = this.synchronizationExecutorService.scheduleWithFixedDelay(() -> {
294            try {
295              if (shouldSynchronize()) {
296                synchronized (eventCache) {
297                  eventCache.synchronize();
298                }
299              }
300            } catch (final RuntimeException runtimeException) {
301              // TODO: log or...?
302              throw runtimeException;
303            }
304          }, 0L, seconds, TimeUnit.SECONDS);
305        assert job != null;
306        this.synchronizationTask = job;
307      }
308      
309    }
310  }

312  /**
313   * Whether, at any given moment, this {@link Reflector} should cause
314   * its {@link EventCache} to {@linkplain EventCache#synchronize()
315   * synchronize}.
316   *
317   * <h2>Design Notes</h2>
318   *
319   * <p>This code follows the Go code in the Kubernetes {@code
320   * client-go/tools/cache} package.  One thing that becomes clear
321   * when looking at all of this through an object-oriented lens is
322   * that it is the {@link EventCache} (the {@code delta_fifo}, in the
323   * Go code) that is ultimately in charge of synchronizing.  It is
324   * not clear why this is a function of a reflector.  In an
325   * object-oriented world, perhaps the {@link EventCache} itself
326   * should be in charge of resynchronization schedules.</p>
327   *
328   * @return {@code true} if this {@link Reflector} should cause its
329   * {@link EventCache} to {@linkplain EventCache#synchronize()
330   * synchronize}; {@code false} otherwise
331   */
332  protected boolean shouldSynchronize() {
333    return this.synchronizationExecutorService != null;
334  }

337    return this.synchronizationInterval;

341    return this.lastSynchronizationResourceVersion;
342  }

348  /**
349   * Using the {@code operation} supplied at construction time,
350   * {@linkplain Listable#list() lists} appropriate Kubernetes
351   * resources, and then, on a separate {@link Thread}, {@linkplain
352   * VersionWatchable sets up a watch} on them, calling {@link
353   * EventCache#replace(Collection, Object)} and {@link
354   * EventCache#add(Object, Event.Type, HasMetadata)} methods as
355   * appropriate.
356   *
357   * <p>The calling {@link Thread} is not blocked by invocations of
358   * this method.</p>
359   *
360   * @see #close()
361   */
362  @NonBlocking
363  public synchronized final void start() {
364    if (this.watch == null) {
365
366      // Run a list operation, and get the resourceVersion of that list.
367      @SuppressWarnings("unchecked")
368      final KubernetesResourceList<? extends T> list = ((Listable<? extends KubernetesResourceList<? extends T>>)this.operation).list();
369      assert list != null;
370      final ListMeta metadata = list.getMetadata();
371      assert metadata != null;      
372      final String resourceVersion = metadata.getResourceVersion();
373      assert resourceVersion != null;
374
375      // Using the results of that list operation, do a full replace
376      // on the EventCache with them.
377      final Collection<? extends T> replacementItems;
378      final Collection<? extends T> items = list.getItems();
379      if (items == null || items.isEmpty()) {
380        replacementItems = Collections.emptySet();
381      } else {
382        replacementItems = Collections.unmodifiableCollection(new ArrayList<>(items));
383      }
384      synchronized (eventCache) {
385        this.eventCache.replace(replacementItems, resourceVersion);
386      }
387
388      // Record the resource version we captured during our list
389      // operation.
390      this.setLastSynchronizationResourceVersion(resourceVersion);
391
392      // Now that we've vetted that our list operation works (i.e. no
393      // syntax errors, no connectivity problems) we can schedule
394      // resynchronizations if necessary.
395      this.setUpSynchronization();
396
397      // Now that we've taken care of our list() operation, set up our
398      // watch() operation.
399      try {
400        @SuppressWarnings("unchecked")
401        final Closeable temp = ((VersionWatchable<? extends Closeable, Watcher<T>>)operation).withResourceVersion(resourceVersion).watch(new WatchHandler());
402        assert temp != null;
403        this.watch = temp;
404      } finally {
405        this.closeSynchronizationExecutorService();
406      }
407    }
408  }

454    /**
455     * Calls the {@link EventCache#add(Object, Event.Type,
456     * HasMetadata)} method on the enclosing {@link Reflector}'s
457     * associated {@link EventCache} with information harvested from
458     * the supplied {@code resource}, and using an {@link Event.Type}
459     * selected appropriately given the supplied {@link
460     * Watcher.Action}.
461     *
462     * @param action the kind of Kubernetes event that happened; must
463     * not be {@code null}
464     *
465     * @param resource the {@link HasMetadata} object that was
466     * affected; must not be {@code null}
467     *
468     * @exception NullPointerException if {@code action} or {@code
469     * resource} was {@code null}
470     *
471     * @exception IllegalStateException if another error occurred
472     */
473    @Override
474    public final void eventReceived(final Watcher.Action action, final T resource) {
475      Objects.requireNonNull(action);
476      Objects.requireNonNull(resource);
477
478      final ObjectMeta metadata = resource.getMetadata();
479      assert metadata != null;
480
481      final Event.Type eventType;
482
483      switch (action) {
484      case ADDED:
485        eventType = Event.Type.ADDITION;
486        break;
487      case MODIFIED:
488        eventType = Event.Type.MODIFICATION;
489        break;
490      case DELETED:
491        eventType = Event.Type.DELETION;
492        break;
493      case ERROR:        
494        // TODO: Uh...the Go code has:
495        //
496        //   if event.Type == watch.Error {
497        //     return apierrs.FromObject(event.Object)
498        //   }
499        //
500        // Now, apierrs.FromObject is here:
501        // https://github.com/kubernetes/apimachinery/blob/kubernetes-1.9.2/pkg/api/errors/errors.go#L80-L88
502        // This is looking for a Status object.  But
503        // WatchConnectionHandler will never forward on such a thing:
504        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L246-L258
505        //
506        // So it follows that if by some chance we get here, resource
507        // will definitely be a HasMetadata.  We go back to the Go
508        // code again, and remember that if the type is Error, the
509        // equivalent of this watch handler simply returns and goes home.
510        //
511        // Now, if we were to throw a RuntimeException here, which is
512        // the idiomatic equivalent of returning and going home, this
513        // would cause a watch reconnect:
514        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L159-L205
515        // ...up to the reconnect limit.
516        //
517        // ...which is fine, but I'm not sure that in an error case a
518        // WatchEvent will ever HAVE a HasMetadata as its payload.
519        // Which means MAYBE we'll never get here.  But if we do, all
520        // we can do is throw a RuntimeException...which ends up
521        // reducing to the same case as the default case below, so we
522        // fall through.
523      default:
524        eventType = null;
525        throw new IllegalStateException();
526      }
527
528      // Add an Event of the proper kind to our EventCache.
529      if (eventType != null) {
530        synchronized (eventCache) {
531          eventCache.add(Reflector.this, eventType, resource);
532        }
533      }
534
535      // Record the most recent resource version we're tracking to be
536      // that of this last successful watch() operation.  We set it
537      // earlier during a list() operation.
538      setLastSynchronizationResourceVersion(metadata.getResourceVersion());
539    }
